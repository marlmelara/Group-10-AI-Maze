\begin{frame}{Task Recap}

\begin{itemize}
\item 64$\times$64 maze, agent has \textbf{no sensor input} - learns only by bumping
\item Hazards: rotating death pits (V-clusters, 90$^{\circ}$ every 5 actions), teleport pads, confusion pads, wind (gamma)
\item Each turn: submit 1--5 actions, receive \texttt{TurnResult}
\item Metrics: success rate, avg turns, avg path length, death rate (+ 4 bonus metrics)
\end{itemize}

\vspace{0.3cm}
\centering
\emph{Train on maze-alpha; test transfer to maze-beta (and gamma for extra credit).}

\end{frame}

%------------------------------------------------
\begin{frame}{Approach: Why Hybrid RL + A*?}

\begin{columns}[T]
\column{0.52\textwidth}

\textbf{Reinforcement Learning}

\begin{itemize}
\item Handles delayed / sparse hazard feedback (you only learn a pit was hot \emph{after} dying)
\item TD(0) updates spread danger to neighbouring cells - agent detours \emph{regions}, not just one cell
\item Hyperparameters transfer between mazes even when cell-specific state doesn't
\end{itemize}

\column{0.48\textwidth}

\textbf{Classical A* Search}

\begin{itemize}
\item Known structure is fully deterministic once discovered - search is optimal
\item Manhattan heuristic keeps 64$\times$64 maze plannable in milliseconds
\item Visit counts + risk penalties become edge weights in A*
\end{itemize}

\end{columns}

\vspace{0.4cm}
\centering
\textbf{Justification:} Spec explicitly asks for RL/EC/ML; RL best fits sparse reward. Classical search guarantees optimal-within-known for discovered regions.

\end{frame}

\begin{frame}{Rotating Fire Hazards}

\begin{columns}[T]
\column{0.5\textwidth}

\textbf{Mechanic}

\begin{itemize}
\item Fires arranged in V-clusters, pivot at the tip
\item Rotate 90$^{\circ}$ clockwise every 5 cumulative actions
\item Agent has no visual - learns only by dying
\end{itemize}

\textbf{Our handling}

\begin{itemize}
\item Parse V-cluster pivots from the source PNG
\item Environment pre-computes all 4 rotation states
\item Agent conservatively treats any cell that has \emph{ever} been a fire as dangerous across all phases
\end{itemize}

\begin{frame}{Mapping and Q-Learning}

\textbf{Per-maze state} (reset between mazes):
\begin{itemize}
\item Trit edge arrays: \texttt{right\_blocked}, \texttt{down\_blocked}
\item \texttt{fires\_by\_phase[4]}, \texttt{confusion\_known}, \texttt{teleport\_map}
\item $Q[r, c, \phi]$ tabular risk, decayed between episodes
\end{itemize}

\textbf{Update rule (on death at $s, \phi$):}

\[
Q[s, \phi] \leftarrow Q[s, \phi] + \alpha\bigl(R_\text{death} - Q[s, \phi]\bigr)
\]

Propagated to Manhattan neighbours with geometric decay $\gamma^{d}$.

\vspace{0.2cm}

\textbf{What transfers between mazes:} the RiskConfig hyperparameters
($\alpha$, $\gamma$, penalty scale, decay factor).  Per-cell values reset.

\end{frame}

%------------------------------------------------
\begin{frame}{Planner: Weighted A*}

\textbf{Edge cost into cell $n$ at phase $\phi$:}

\[
c(n, \phi) =
\underbrace{1}_{\text{step}} +
\underbrace{w_\text{fire} \mathbb{1}[n \in F_\phi]}_{\text{fire risk}} +
\underbrace{w_\text{conf} \mathbb{1}[n \in C]}_{\text{confusion}} +
\underbrace{w_Q Q[n, \phi]}_{\text{learned}} +
\underbrace{\rho \cdot v[n]}_{\text{visit}}
\]

\vspace{0.1cm}

\textbf{Two-stage search:}
\begin{enumerate}
\item Strict pass - hard-block known fires (any phase)
\item Relaxed pass (only if strict fails) - allow traversal with high cost
\item Frontier fallback - if goal unreachable, head toward nearest unknown edge
\end{enumerate}

\vspace{0.2cm}

\textbf{Batching:} submit actions while each next edge is known-open
and landing cell is known-safe at that phase; first uncertain step
ends the batch.

\end{frame}

\begin{frame}{Extra Credit: maze-gamma Wind Hazard}

\begin{columns}[T]
\column{0.5\textwidth}

\textbf{Wind cells ($\uparrow, \leftarrow$)}

\begin{itemize}
\item Impassable - bumping deflects you one cell in the arrow direction
\item The rest of the current turn's actions become moves in the wind direction
\end{itemize}

\textbf{Adaptation}

\begin{itemize}
\item Treat the edge \emph{into} a wind cell as a wall in our map
\item Once discovered, planner routes around
\item No new learning machinery needed
\end{itemize}

\begin{frame}{Conclusion \& Lessons Learned}

\begin{itemize}
\item Hybrid RL + A* reaches the spec's expected benchmarks on alpha and beta
\item Conservative fire avoidance (mark-all-phases-on-death) cut regressions between episodes
\item Cached-successful-path retrace dramatically reduced replanning cost after the first success
\item Wind hazard handled as a wall class - no new architecture
\end{itemize}
